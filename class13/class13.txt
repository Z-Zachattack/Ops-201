#!/bin/bash

# Function to perform whois, dig, host, and nslookup and save the results to a .txt file
perform_network_checks() {
    read -p "Enter a domain name (e.g., Google.com): " domain_name

    # Perform whois
    whois "$domain_name" > "$domain_name-results.txt"

    # Perform dig
    dig "$domain_name" >> "$domain_name-results.txt"

    # Perform host
    host "$domain_name" >> "$domain_name-results.txt"

    # Perform nslookup
    nslookup "$domain_name" >> "$domain_name-results.txt"

    echo "Results have been saved to $domain_name-results.txt"
}

# Main menu
echo "Select an option:"
echo "1. Option 1"
echo "2. Option 2"
echo "3. Option 3"
echo "4. Option 4"
echo "5. Option 5"
echo "6. Perform Network Checks"

read -p "Enter your choice: " choice

case $choice in
    1)
        # Option 1 code here
        ;;
    2)
        # Option 2 code here
        ;;
    3)
        # Option 3 code here
        ;;
    4)
        # Option 4 code here
        ;;
    5)
        # Option 5 code here
        ;;
    6)
        # Call the network checks function
        perform_network_checks
        ;;
    *)
        echo "Invalid choice"
        ;;
esac
